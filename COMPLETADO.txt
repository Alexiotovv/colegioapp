💡 EJEMPLOS DE USO
═════════════════════════════════════════════════════════════════════════════════

CREAR EVALUACIÓN LITERAL:
   Evaluacion::create([
       'nombre' => 'Evaluación de Comportamiento',
       'etiqueta_libreta' => 'Comportamiento',
       'tipo_escala' => 'LITERAL',
       'opciones_literales' => json_encode(['AD', 'A', 'B', 'C']),
       'nivel_id' => 1,
       'visible_primaria' => true,
       'requerido' => true,  // 🔴
   ]);

CREAR EVALUACIÓN NUMÉRICA:
   Evaluacion::create([
       'nombre' => 'Examen Final',
       'etiqueta_libreta' => 'Examen',
       'tipo_escala' => 'NUMERICO',
       'escala_minima' => 0,
       'escala_maxima' => 20,
       'nivel_id' => 2,
       'visible_secundaria' => true,
       'requerido' => true,  // 🔴
   ]);

VALIDAR VALORES:
   $evaluacion->esValorValido('AD');       // true
   $evaluacion->getOpcionesArray();        // ['AD', 'A', 'B', 'C']
   $evaluacion->esVisiblePara('Primaria'); // true
